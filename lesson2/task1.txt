var a = 1, b = 1, c, d;
c = ++a; alert(c);           // 2 - тут все понятно, присваиваем переменной С инкремированое значение пременной А (т.е. а == 2 теперь)
d = b++; alert(d);           // 1 - тут постфиксная форма, поэтому сначала идет присвоение потом инкремент, т.о. b == 2 после этой строки
c = (2+ ++a); alert(c);      // 5 - мы уже увеличили а до 2 и теперь увеличили еще раз до 3 и прибавили к 2 - получили 5. a при этом стало == 3
d = (2+ b++); alert(d);      // 4 - то же что и в предыдущей строке но постфиксная форма, опять сначала использовали старое значение b для выражения а потом увеличили, теперь b == 3
alert(a);                    // 3 написано выше
alert(b);                    // 3 написано выше
